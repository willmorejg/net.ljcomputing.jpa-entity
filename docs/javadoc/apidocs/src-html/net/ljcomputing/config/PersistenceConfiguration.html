017package net.ljcomputing.config;
018
019import java.sql.Driver;
020import java.sql.DriverManager;
021import java.sql.SQLException;
022
023import javax.persistence.EntityManagerFactory;
024import javax.sql.DataSource;
025
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.springframework.beans.factory.annotation.Value;
029import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
030import org.springframework.boot.autoconfigure.domain.EntityScan;
031import org.springframework.context.annotation.Bean;
032import org.springframework.context.annotation.ComponentScan;
033import org.springframework.context.annotation.Configuration;
034import org.springframework.context.annotation.EnableAspectJAutoProxy;
035import org.springframework.context.annotation.PropertySource;
036import org.springframework.context.support.PropertySourcesPlaceholderConfigurer;
037import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
038import org.springframework.orm.jpa.JpaTransactionManager;
039import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
040import org.springframework.orm.jpa.vendor.HibernateJpaVendorAdapter;
041import org.springframework.transaction.PlatformTransactionManager;
042import org.springframework.transaction.annotation.EnableTransactionManagement;
043
044import com.zaxxer.hikari.HikariConfig;
045import com.zaxxer.hikari.HikariDataSource;
046
047/**
048 * Persistence configuration.
049 * 
050 * @author James G. Willmore
051 *
052 */
053@Configuration
054@PropertySource("classpath:datasource.properties")
055@ComponentScan(basePackages = { "net.ljcomputing" })
056@EnableAutoConfiguration
057@EnableAspectJAutoProxy
058@EnableJpaRepositories(basePackages = { "net.ljcomputing.repository" })
059@EntityScan({ "net.ljcomputing.entity" })
060@EnableTransactionManagement
061@SuppressWarnings("PMD.AtLeastOneConstructor")
062public class PersistenceConfiguration {
063
064  /** The SLF4J Logger. */
065  private static final Logger LOGGER = LoggerFactory.getLogger(PersistenceConfiguration.class);
066
067  /** The url. */
068  @Value("${datasource.url}")
069  private transient String url;
070
071  /** The username. */
072  @Value("${datasource.username}")
073  private transient String username;
074
075  /** The password. */
076  @Value("${datasource.password}")
077  private transient String password;
078
079  /** The pool name. */
080  @Value("${datasource.poolName}")
081  private transient String poolName;
082
083  /** The connection timeout. */
084  @Value("${datasource.connectionTimeout}")
085  private transient int connectionTimeout;
086
087  /** The max lifetime. */
088  @Value("${datasource.maxLifetime}")
089  private transient int maxLifetime;
090
091  /** The maximum pool size. */
092  @Value("${datasource.maximumPoolSize}")
093  private transient int maximumPoolSize;
094
095  /** The minimum idle. */
096  @Value("${datasource.minimumIdle}")
097  private transient int minimumIdle;
098
099  /** The idle timeout. */
100  @Value("${datasource.idleTimeout}")
101  private transient int idleTimeout;
102
103  /**
104   * Property sources placeholder configurer.
105   *
106   * @return the property sources placeholder configurer
107   */
108  @Bean
109  public static PropertySourcesPlaceholderConfigurer propertySourcesPlaceholderConfigurer() {
110    return new PropertySourcesPlaceholderConfigurer();
111  }
112
113  /**
114   * Data source.
115   *
116   * @return the data source
117   */
118  @Bean(destroyMethod = "close")
119  public DataSource dataSource() {
120    LOGGER.debug("using the following configuration: url:[{}] username:[{}] password:[********]", url, username);
121
122    try {
123      return new HikariDataSource(hikariConfig());
124    } catch (SQLException exception) {
125      LOGGER.error("Could not create data source:", exception);
126    }
127
128    return null;
129  }
130
131  /**
132   * Gets the driver class name.
133   *
134   * @return the driver class name
135   * @throws SQLException the SQL exception
136   */
137  private String getDriverClassName() throws SQLException {
138    final Driver driver = DriverManager.getDriver(url);
139    final Class<?> driverClass = driver.getClass();
140    return driverClass.getName();
141  }
142
143  /**
144   * Hikari config.
145   *
146   * @return the hikari config
147   * @throws SQLException the SQL exception
148   */
149  @Bean
150  public HikariConfig hikariConfig() throws SQLException {
151    final HikariConfig hikariConfig = new HikariConfig();
152
153    initConnectionPool(hikariConfig);
154
155    hikariConfig.setPoolName(poolName);
156    hikariConfig.setMaximumPoolSize(maximumPoolSize);
157    hikariConfig.setMinimumIdle(minimumIdle);
158    hikariConfig.setMaxLifetime(maxLifetime);
159    hikariConfig.setConnectionTimeout(connectionTimeout);
160    hikariConfig.setIdleTimeout(idleTimeout);
161
162    addAdditionalPoolProperties(hikariConfig);
163
164    return hikariConfig;
165  }
166
167  /**
168   * Inits the connection pool.
169   *
170   * @param hikariConfig the hikari config
171   * @throws SQLException the SQL exception
172   */
173  private void initConnectionPool(final HikariConfig hikariConfig) throws SQLException {
174    hikariConfig.setDriverClassName(getDriverClassName());
175    hikariConfig.setJdbcUrl(url);
176    hikariConfig.setUsername(username);
177    hikariConfig.setPassword(password);
178  }
179
180  /**
181   * Adds the additional connection pool properties.
182   *
183   * @param hikariConfig the hikari config
184   */
185  private void addAdditionalPoolProperties(final HikariConfig hikariConfig) {
186    hikariConfig.addDataSourceProperty("datasource.cachePrepStmts", "true");
187    hikariConfig.addDataSourceProperty("datasource.prepStmtCacheSize", "250");
188    hikariConfig.addDataSourceProperty("datasource.prepStmtCacheSqlLimit", "2048");
189    hikariConfig.addDataSourceProperty("datasource.useServerPrepStmts", "true");
190  }
191
192  /**
193   * Entity manager factory.
194   *
195   * @return the entity manager factory
196   */
197  @Bean
198  public EntityManagerFactory entityManagerFactory() {
199    final HibernateJpaVendorAdapter vendorAdapter = new HibernateJpaVendorAdapter();
200    vendorAdapter.setGenerateDdl(true);
201
202    final LocalContainerEntityManagerFactoryBean factory = new LocalContainerEntityManagerFactoryBean();
203    factory.setJpaVendorAdapter(vendorAdapter);
204    factory.setPackagesToScan("net.ljcomputing.entity");
205    factory.setDataSource(dataSource());
206    factory.afterPropertiesSet();
207
208    return factory.getObject();
209  }
210
211  /**
212   * Transaction manager.
213   *
214   * @return the data source transaction manager
215   */
216  @Bean
217  public PlatformTransactionManager transactionManager() {
218    final JpaTransactionManager txManager = new JpaTransactionManager();
219    txManager.setEntityManagerFactory(entityManagerFactory());
220    return txManager;
221  }
222}
